import yfinance as yf from arch import arch_model import re, datetime, pytz, warnings import pandas as pd import numpy as np warnings.filterwarnings("ignore") melbourne_tz = pytz.timezone('Australia/Melbourne') # لیست دارایی‌ها assets = { 'btc': 'BTC-USD', 'eth': 'ETH-USD', 'sol': 'SOL-USD', 'gold': 'GC=F', 'silver': 'SI=F', 'sp500': '^GSPC', 'nsdq100': '^IXIC', 'dj30': '^DJI', 'rty2000': '^RUT', 'eurusd': 'EURUSD=X', 'usdjpy': 'JPY=X', 'audjpy': 'AUDJPY=X' } def start_analysis(): now = datetime.datetime.now(melbourne_tz) print(f"🚀 شروع تحلیل | ملبورن: {now.strftime('%H:%M:%S')}") try: with open("index.html", "r", encoding="utf-8") as f: html = f.read() except FileNotFoundError: print("❌ فایل پیدا نشد!"); return # ۱. آپدیت زمان ملبورن html = re.sub(r'Melbourne:.*?(?=<)', f'Melbourne: {now.strftime("%d/%m/%Y, %H:%M:%S")}', html, count=1) for label, ticker in assets.items(): try: data = yf.download(ticker, period="1mo", interval="1d", progress=False) if data.empty: continue last_price = float(data['Close'].iloc[-1]) returns = 100 * data['Close'].pct_change().dropna() # محاسبه نوسان model = arch_model(returns, vol='Garch', p=1, q=1, dist='studentst') res = model.fit(disp='off') vol_val = np.sqrt(res.forecast(horizon=1).variance.values[-1, 0]) # --- اصلاح اصلی اینجاست: استفاده از count=1 --- # جایگزینی قیمت مخصوص هر نماد price_id = f'id="price-{label}"' html = re.sub(rf'{price_id}>[\d\.,-]+', f'{price_id}>{last_price:,.2f}', html, count=1) # جایگزینی مقدار GARCH مخصوص هر نماد garch_id = f'id="garch-{label}"' html = re.sub(rf'{garch_id}>[\d\.,%]+', f'{garch_id}>{vol_val:.2f}%', html, count=1) print(f"✅ {label.upper()} بروزرسانی شد: {last_price:,.2f}") except Exception as e: print(f"⚠️ خطا در {label}: {e}") with open("index.html", "w", encoding="utf-8") as f: f.write(html) print("\n✨ اصلاح نهایی انجام شد. حالا فایل را در گیت‌هاب آپلود کنید.") if __name__ == "__main__": start_analysis()
